00105 q15_t arm_cos_q15(
00106   q15_t x)
00107 {
00108   q31_t cosVal;                                  /* Temporary variable for output */
00109   q15_t *tablePtr;                               /* Pointer to table */
00110   q15_t in, in2;                                 /* Temporary variables for input */
00111   q31_t wa, wb, wc, wd;                          /* Cubic interpolation coefficients */
00112   q15_t a, b, c, d;                              /* Four nearest output values */
00113   q15_t fract, fractCube, fractSquare;           /* Variables for fractional value */
00114   q15_t oneBy6 = 0x1555;                         /* Fixed point value of 1/6 */
00115   q15_t tableSpacing = TABLE_SPACING_Q15;        /* Table spacing */
00116   int32_t index;                                 /* Index variable */
00117 
00118   in = x;
00119 
00120   /* Calculate the nearest index */
00121   index = (int32_t) in / tableSpacing;
00122 
00123   /* Calculate the nearest value of input */
00124   in2 = (q15_t) index *tableSpacing;
00125 
00126   /* Calculation of fractional value */
00127   fract = (in - in2) << 8;
00128 
00129   /* fractSquare = fract * fract */
00130   fractSquare = (q15_t) ((fract * fract) >> 15);
00131 
00132   /* fractCube = fract * fract * fract */
00133   fractCube = (q15_t) ((fractSquare * fract) >> 15);
00134 
00135   /* Initialise table pointer */
00136   tablePtr = (q15_t *) & cosTableQ15[index];
00137 
00138   /* Cubic interpolation process */
00139   /* Calculation of wa */
00140   /* wa = -(oneBy6)*fractCube + (fractSquare >> 1u) - (0x2AAA)*fract; */
00141   wa = (q31_t) oneBy6 *fractCube;
00142   wa += (q31_t) 0x2AAA *fract;
00143   wa = -(wa >> 15);
00144   wa += (fractSquare >> 1u);
00145 
00146   /* Read first nearest value of output from the cos table */
00147   a = *tablePtr++;
00148 
00149   /* cosVal = a * wa */
00150   cosVal = a * wa;
00151 
00152   /* Calculation of wb */
00153   wb = (((fractCube >> 1u) - fractSquare) - (fract >> 1u)) + 0x7FFF;
00154 
00155   /* Read second nearest value of output from the cos table */
00156   b = *tablePtr++;
00157 
00158   /*      cosVal += b*wb */
00159   cosVal += b * wb;
00160 
00161   /* Calculation of wc */
00162   wc = -(q31_t) fractCube + fractSquare;
00163   wc = (wc >> 1u) + fract;
00164 
00165   /* Read third nearest value of output from the cos table */
00166   c = *tablePtr++;
00167 
00168   /*      cosVal += c*wc */
00169   cosVal += c * wc;
00170 
00171   /* Calculation of wd */
00172   /* wd = (oneBy6)*fractCube - (oneBy6)*fract; */
00173   fractCube = fractCube - fract;
00174   wd = ((q15_t) (((q31_t) oneBy6 * fractCube) >> 15));
00175 
00176   /* Read fourth nearest value of output from the cos table */
00177   d = *tablePtr++;
00178 
00179   /* cosVal += d*wd; */
00180   cosVal += d * wd;
00181 
00182   /* Return the output value in 1.15(q15) format */
00183   return ((q15_t) (cosVal >> 15u));
